# CCP4-Specific Packages (symlinked from CCP4-20251105, not pip-installed):
# - clipper: Crystallographic library (symlinked from CCP4)
# - iris_validation: Protein validation metrics (symlinked from CCP4)
# - pyrvapi: Report Viewing API (symlinked from CCP4)
# - ccp4mg: CCP4 molecular graphics library (includes mmut module)
# - phaser: Phasing software package
# - mrbump: Molecular replacement pipeline
# - chem_data: Top8000 Ramachandran/rotamer database (for validation)
# - mmut: mmCIF utilities (stub from stubs/mmut.py.stub)
#
# CCTBX Suite (symlinked from CCP4-20251105 for version 2025.11.dev16):
# - cctbx: Core crystallography toolkit
# - mmtbx: Macromolecular crystallography toolkit
# - iotbx: I/O toolbox for crystallographic formats
# - scitbx: Scientific computing toolbox
# - libtbx: Low-level toolbox utilities
# - smtbx: Small molecule crystallography toolkit
# - boost_adaptbx: Boost Python adapters
# - share/cctbx: Build environment configuration
#
# To symlink these modules (Python 3.11 venv):
#   CCP4_SITE=/Users/nmemn/Developer/ccp4-20251105/Frameworks/Python.framework/Versions/3.11/lib/python3.11/site-packages
#   VENV_SITE=.venv.py311/lib/python3.11/site-packages

#   # Core CCP4 modules
#   ln -sf $CCP4_SITE/{clipper.py,_clipper.so,iris_validation,pyrvapi.so,pyrvapi_ext,ccp4mg,phaser,mrbump,chem_data} $VENV_SITE/
#
#   # CCTBX suite
#   ln -sf $CCP4_SITE/{cctbx,mmtbx,iotbx,scitbx,libtbx,smtbx,boost_adaptbx} $VENV_SITE/
#   mkdir -p .venv.py311/share
#   ln -sf /Users/nmemn/Developer/ccp4-20251105/Frameworks/Python.framework/Versions/3.11/share/cctbx .venv.py311/share/cctbx
